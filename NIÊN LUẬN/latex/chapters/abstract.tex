\chapter*{Tóm tắt}
\addcontentsline{toc}{chapter}{Tóm tắt}

\textit{Viết tóm tắt ngắn gọn về toàn bộ đề tài. Tóm tắt cần bao gồm:}
\begin{itemize}
    \item Mục tiêu chính của đề tài.
    \item Mô tả ngắn gọn phương pháp / cách tiếp cận.
    \item Kết quả chính hoặc phát hiện từ đề tài.
    \item Ý nghĩa hoặc tác động của đề tài.
    \item Độ dài: 150–200 từ, viết sau cùng nhưng trình bày ở đầu.
\end{itemize}
