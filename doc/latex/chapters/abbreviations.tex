\chapter*{Danh mục các từ viết tắt}
\addcontentsline{toc}{chapter}{Danh mục các từ viết tắt}

\begin{tabularx}{\textwidth}{|l|X|}
\hline
\textbf{Từ viết tắt} & \textbf{Giải thích đầy đủ} \\ \hline
2FA & Two-Factor Authentication --- Xác thực hai yếu tố \\ \hline
ACID & Atomicity, Consistency, Isolation, Durability --- Tính nguyên tử, nhất quán, cô lập, bền vững \\ \hline
AES & Advanced Encryption Standard --- Tiêu chuẩn mã hóa nâng cao \\ \hline
AI & Artificial Intelligence --- Trí tuệ nhân tạo \\ \hline
API & Application Programming Interface --- Giao diện lập trình ứng dụng \\ \hline
CRUD & Create, Read, Update, Delete --- Tạo, Đọc, Cập nhật, Xóa \\ \hline
CSV & Comma-Separated Values --- Giá trị phân cách bằng dấu phẩy \\ \hline
ERD & Entity Relationship Diagram --- Sơ đồ quan hệ thực thể \\ \hline
FR & Functional Requirement --- Yêu cầu chức năng \\ \hline
HTTP & Hypertext Transfer Protocol --- Giao thức truyền siêu văn bản \\ \hline
HTTPS & Hypertext Transfer Protocol Secure --- Giao thức truyền siêu văn bản bảo mật \\ \hline
IoC & Inversion of Control --- Đảo ngược điều khiển \\ \hline
JWT & JSON Web Token \\ \hline
LLM & Large Language Model --- Mô hình ngôn ngữ lớn \\ \hline
MVP & Minimum Viable Product --- Sản phẩm khả thi tối thiểu \\ \hline
NFR & Non-Functional Requirement --- Yêu cầu phi chức năng \\ \hline
NLP & Natural Language Processing --- Xử lý ngôn ngữ tự nhiên \\ \hline
OAuth & Open Authorization --- Ủy quyền mở \\ \hline
OCR & Optical Character Recognition --- Nhận dạng ký tự quang học \\ \hline
PDF & Portable Document Format \\ \hline
RDBMS & Relational Database Management System --- Hệ quản trị cơ sở dữ liệu quan hệ \\ \hline
REST & Representational State Transfer \\ \hline
SDK & Software Development Kit --- Bộ công cụ phát triển phần mềm \\ \hline
SQL & Structured Query Language --- Ngôn ngữ truy vấn có cấu trúc \\ \hline
SRS & Software Requirement Specification --- Đặc tả yêu cầu phần mềm \\ \hline
STT & Speech-to-Text --- Chuyển giọng nói thành văn bản \\ \hline
TLS & Transport Layer Security --- Bảo mật tầng vận chuyển \\ \hline
UAT & User Acceptance Testing --- Kiểm thử chấp nhận người dùng \\ \hline
UI & User Interface --- Giao diện người dùng \\ \hline
UX & User Experience --- Trải nghiệm người dùng \\ \hline
\end{tabularx}
