\section{Kết quả đạt được}
\label{sec:achievements}

\textit{Phần này sẽ được viết sau khi hoàn thành phát triển ứng dụng, tóm tắt các mục tiêu đã đạt được so với Chương 1 (Giới thiệu), bao gồm:}

\begin{itemize}
    \item Mức độ hoàn thành hệ thống nhập liệu đa phương thức bằng AI (văn bản, giọng nói, hình ảnh).
    \item Hiệu quả của hệ thống phân loại giao dịch thông minh.
    \item Tính năng quản lý ngân sách và cảnh báo chủ động.
    \item Khả năng phân tích và báo cáo cá nhân hóa.
    \item Hệ thống quản lý đa tài khoản và dòng tiền.
    \item Trải nghiệm người dùng với nhân cách AI.
    \item Bảo mật và xuất dữ liệu.
\end{itemize}

\section{Hạn chế}
\label{sec:limitations}

\textit{Phần này sẽ đánh giá các hạn chế thực tế của phiên bản hiện tại sau khi kiểm thử, bao gồm:}

\begin{itemize}
    \item Giới hạn về thời gian phát triển trong khuôn khổ niên luận.
    \item Các tính năng chưa hoàn thiện hoặc cần cải tiến.
    \item Dữ liệu kiểm thử và quy mô người dùng thử nghiệm hạn chế.
    \item Phụ thuộc vào dịch vụ AI bên ngoài (API cloud).
\end{itemize}

\section{Hướng phát triển}
\label{sec:future_work}

\textit{Phần này sẽ đề xuất hướng phát triển trong tương lai dựa trên kết quả thực tế, dự kiến bao gồm:}

\begin{itemize}
    \item Mở rộng hỗ trợ đa ngôn ngữ (ngoài Việt --- Anh).
    \item Tích hợp API ngân hàng để đồng bộ giao dịch tự động (Open Banking).
    \item Phát triển tính năng ví chung (Shared Wallets) cho nhóm.
    \item Tự động đề xuất hạn mức ngân sách dựa trên lịch sử chi tiêu (AI auto-suggest).
    \item Tối ưu hóa model AI on-device để giảm phụ thuộc vào cloud.
\end{itemize}
